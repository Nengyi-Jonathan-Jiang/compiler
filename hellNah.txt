double truncate(double a, int c) {
    string s = to_string(a);
    bool start = false; string ret = "";
    for (auto i : s) {
        if(c == 0){if(i=='.'){return stod(ret);}}
        if (i == '.') { start = true; ret += i; }
        else {
            ret += i;
            if (start) { c--; if (c == 0) { return stod(ret); } }
        }
    }
    return stod(ret);
}
